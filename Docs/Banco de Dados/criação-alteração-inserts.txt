CREATE TABLE `alugado` (
 `idaluguel` int(10) unsigned NOT NULL AUTO_INCREMENT,
 `usuario_idusuario` int(10) unsigned NOT NULL,
 `produto_idproduto` int(10) unsigned NOT NULL,
 `data_inicio_alugado` date NOT NULL,
 `data_fim_alugado` date NOT NULL,
 PRIMARY KEY (`idaluguel`),
 KEY `aluguel_FKIndex1` (`produto_idproduto`),
 KEY `aluguel_FKIndex3` (`usuario_idusuario`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1

CREATE TABLE `cidade` (
 `idcidade` int(10) unsigned NOT NULL AUTO_INCREMENT,
 `estado_pais_idpais` int(10) unsigned NOT NULL,
 `estado_idestado` int(10) unsigned NOT NULL,
 `nome_cidade` varchar(255) NOT NULL,
 PRIMARY KEY (`idcidade`,`estado_pais_idpais`,`estado_idestado`),
 KEY `cidade_FKIndex1` (`estado_idestado`,`estado_pais_idpais`)
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=latin1

CREATE TABLE `estado` (
 `estado` int(10) unsigned NOT NULL AUTO_INCREMENT,
 `pais` int(10) unsigned NOT NULL,
 `nome_estado` varchar(255) NOT NULL,
 `sigla_estado` varchar(2) NOT NULL,
 PRIMARY KEY (`estado`,`pais`),
 KEY `estado_FKIndex1` (`pais`)
) ENGINE=InnoDB AUTO_INCREMENT=4 DEFAULT CHARSET=latin1

CREATE TABLE `fornecido` (
 `idaluguelfornecedor` int(10) unsigned NOT NULL AUTO_INCREMENT,

 `usuario` int(10) unsigned NOT NULL,

 `data_fim_fornecido` date NOT NULL,
 PRIMARY KEY (`idaluguelfornecedor`),
 KEY `aluguelFornecedor_FKIndex1` (`usuario`),
 KEY `fornecidos_FKIndex2` (`produto`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1

CREATE TABLE `pais` (

 `nome_pais` varchar(45) NOT NULL,

) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=latin1

CREATE TABLE `produto` (

 `nome_produto` varchar(255) NOT NULL,

 `localizacao` int(10) NOT NULL,
